#include <cstdio>
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

void solution(string n1, string n2){
    string ans = "";
    reverse(n1.begin(),n1.end());
    reverse(n2.begin(),n2.end());
    if (n1.size() > n2.size()){
        int diff = n1.size() - n2.size();
        for (int i = 0; i < diff; ++i) {
            n2 += "0";
        }
    }else{
        int diff = n2.size() - n1.size();
        for (int i = 0; i < diff; ++i) {
            n1 += "0";
        }
    }

    int carry = 0;
    for (int i = 0; i < n1.size(); ++i) {
        int sum = (n1[i]-'0') + (n2[i]-'0') + carry;
        if (sum >= 10){
            carry = 1;
            ans += (sum%10) + '0';
        }else{
            carry = 0;
            ans += sum + '0';
        }
    }
    if (carry == 1) ans += '1';

    reverse(ans.begin(), ans.end());
    printf("%s\n", ans.c_str());
}

int main() {
    string n1, n2;
    cin >> n1 >> n2;

    solution(n1, n2);

    return 0;
}
