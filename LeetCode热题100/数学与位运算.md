# 数学与位运算

## 1.只出现一次的数字
```c++
class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int ans = 0;
        for (int num : nums) {
            ans ^= num;
        }
        return ans;
    }
};
```

## 2.多数元素
```c++
class Solution {
public:
    int majorityElement(vector<int>& nums) {
        // 计数器
        int ans = 0;
        int cnt = 0;
        for (int num : nums) {
            if (cnt == 0) {
                ans = num;
                cnt += 1;
            } else if (cnt > 0) {
                cnt += num == ans ? 1 : -1;
            }
        }
        return ans;
    }
};
```
