#include <iostream>
using namespace std;

int q[100010];

int main(){
    int front = 0;
    int back = -1;
    
    int n;
    cin >> n;
    
    for (int i = 0; i < n; i++){
        string op;
        cin >> op;
        if (op == "push"){
            int val;
            scanf("%d", &val);
            q[++back] = val;
        }else if (op == "pop"){
            front++;
        }else if (op == "empty"){
            printf("%s\n", front > back ? "YES" : "NO");
        }else if (op == "query"){
            printf("%d\n", q[front]);
        }
    }
    
    return 0;
}
