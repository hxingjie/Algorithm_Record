# 高精度减法
```c++
#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    string str1, str2;
    cin >> str1 >> str2;
    // 比较大小，让str1存储较大值
    bool isPositive = true;
    if (str1.size() < str2.size() || (str1.size() == str2.size() && str1 < str2)) {
        isPositive = false;
        string str = str1;
        str1 = str2;
        str2 = str;
    }

    vector<int> num1, num2;
    for (int i = str1.size()-1; i >= 0; --i)
        num1.push_back(str1[i]-'0');
    for (int i = str2.size()-1; i >= 0; --i)
        num2.push_back(str2[i]-'0');

    vector<int> ans;
    int borrow = 0;// 借位
    for (int i = 0; i < num1.size(); ++i) {
        int temp = num1[i] - borrow;
        if (i < num2.size()) temp -= num2[i];
        if (temp >= 0){
            ans.push_back(temp);
            borrow = 0;
        }else{
            ans.push_back(temp+10);
            borrow = 1;
        }
    }
    while (ans.size() > 1 && ans.back() == 0) ans.pop_back();// 去除前导0

    if (!isPositive) printf("-");
    for (int i = ans.size()-1; i >= 0; --i) {
        printf("%d", ans[i]);
    }
    printf("\n");

    return 0;
}
```
