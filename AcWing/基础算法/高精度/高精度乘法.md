# 高精度乘法
```
#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
    string str;
    cin >> str;
    
    int num2;
    cin >> num2;
    if (num2 == 0){
        printf("0\n");
        return 0;
    }
    
    vector<int> num1;
    for (int i = str.size()-1; i >= 0; --i)
        num1.push_back(str[i]-'0');
        
    int carry = 0;
    vector<int> ans;
    for (int i = 0; i < num1.size() || carry > 0; ++i) {
        // num1处理完后，还需要处理carry，与加法不同，此处的carry可能是大于或等于10的数，需要循环取数
        int temp = carry;
        if (i < num1.size()) temp += num1[i] * num2;
        ans.push_back(temp % 10);
        carry = temp / 10;
    }

    for (int i = ans.size()-1; i >= 0; --i)
        printf("%d", ans[i]);
        
    printf("\n");

    return 0;
}
```
