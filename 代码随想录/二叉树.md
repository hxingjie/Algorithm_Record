```c++
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> ans;

        queue<TreeNode*> sup_q;

        if (root == nullptr)
            return ans;
        sup_q.push(root);
        while (!sup_q.empty()){
            int sz = sup_q.size();
            vector<int> tmp;
            for (int i = 0; i < sz; i++){
                TreeNode* cur = sup_q.front();
                sup_q.pop();
                tmp.push_back(cur->val);
                if (cur->left != nullptr)
                    sup_q.push(cur->left);
                if (cur->right != nullptr)
                    sup_q.push(cur->right);
            }
            ans.push_back(tmp);
        }
        return ans;
    }
};
```

```c++
class Solution {
public:
    bool sup_func(TreeNode* left, TreeNode* right){
        if ((left == nullptr && right != nullptr)
            || (left != nullptr && right == nullptr))
            return false;
        
        if (left == nullptr && right == nullptr)
            return true;

        if (left->val == right->val)
            return sup_func(left->left, right->right) 
                       && sup_func(left->right, right->left);
        else
            return false;
        
    }
    bool isSymmetric(TreeNode* root) {
        return sup_func(root->left, root->right);
    }
};
```

### (4)二叉树的所有路径

https://leetcode.cn/problems/binary-tree-paths/description/

```c++
class Solution {
public:
    vector<vector<int>> ans;
    vector<int> track;
    string num2str(int num){
        if (num == 0)
            return "0";
        
        string s = "";
        int isNeg = 0;
        if (num < 0){
            isNeg = 1;
            num = -num;
            s.push_back('-');
        }
        
        while (num > 0){
            s.push_back((num % 10) + '0');
            num /= 10;
        }
        reverse(s.begin()+isNeg, s.end());
        return s;
    }

    void traverse(TreeNode* root){
        if (root == nullptr)
            return;
        
        track.push_back(root->val);

        if (root->left == nullptr && root->right == nullptr){
            ans.push_back(track);
            track.pop_back();
            return;
        }

        traverse(root->left);
        traverse(root->right);

        track.pop_back();

    }

    vector<string> binaryTreePaths(TreeNode* root) {
        traverse(root);
        vector<string> res;

        for (vector<int> v : ans){
            string tmp = "";
            for (int i = 0; i < v.size(); i++){
                tmp += num2str(v[i]);
                if (i < v.size()-1)
                    tmp += "->";
            }
            res.push_back(tmp);
        }
        return res;
    }
};
```

### (7)路径总和

https://leetcode.cn/problems/path-sum/description/

```c++
class Solution {
public:
    int target;
    int trace = 0;
    bool isFound = false;
    void backtrack(TreeNode* root){
        if (isFound)
            return;

        if (root == nullptr)
            return;

        trace += root->val;
        if (trace == target 
            && root->left == nullptr && root->right == nullptr){
            isFound = true;
            trace -= root->val;
            return;
        }

        backtrack(root->left);
        backtrack(root->right);

        trace -= root->val;
    }
    bool hasPathSum(TreeNode* root, int targetSum) {
        target = targetSum;
        backtrack(root);
        return isFound;
    }
};
```

```c++
class Solution {
public:
    TreeNode* buildTree(vector<int>& inorder, vector<int>& postorder) {
        if (inorder.size() <= 0)
            return nullptr;
        
        int root_val = postorder.back();
        TreeNode* root = new TreeNode(root_val, nullptr, nullptr);
        int idx = -1;
        for (int i = 0; i < inorder.size(); i++)
            if (inorder[i] == root_val)
                idx = i;
        
        // 0 -> idx-1, idx+1 -> size-1
        // 0 -> idx-1, idx -> size-2
        vector<int> left_in = vector<int>(inorder.begin(), inorder.begin()+idx);
        vector<int> left_post = vector<int>(postorder.begin(), postorder.begin()+idx);

        vector<int> right_in = vector<int>(inorder.begin()+idx+1, inorder.end());
        vector<int> riight_post = vector<int>(postorder.begin()+idx, postorder.end()-1);
        
        root->left = buildTree(left_in, left_post);
        root->right = buildTree(right_in, riight_post);

        return root;
    }
};
```

```c++
class Solution {
public:
    TreeNode* preNode;
    int curCnt;
    int maxCnt;
    vector<int> ans;
    void traverse(TreeNode* root){
        if (root == nullptr)
            return;
        
        traverse(root->left);
        if (preNode != nullptr){
            if (root->val == preNode->val){
                curCnt++;
                if (curCnt == maxCnt){
                    ans.push_back(root->val);
                }
                else if (curCnt > maxCnt){
                    maxCnt = curCnt;
                    ans.clear();
                    ans.push_back(root->val);
                }

                curCnt = 1;
                if (curCnt == maxCnt){
                    ans.push_back(root->val);
                }
            }
        }else{// 第一个遍历的节点
            curCnt = 1;
            maxCnt = 1;
            ans.push_back(root->val);
        }
        preNode = root;
        traverse(root->right);
    }
    vector<int> findMode(TreeNode* root) {
        traverse(root);
        return ans;
    }
};
```

### (1)二叉树的最近公共祖先

```c++
class Solution {
public:
    TreeNode* findNode(TreeNode* root, TreeNode* child){
        if (root == nullptr)
            return nullptr;

        if (root == child)
            return root;

        TreeNode* l = findNode(root->left, child);
        if (l != nullptr)

        TreeNode* r = findNode(root->right, child);
        if (r != nullptr)
            return r;
        
        return nullptr;
    }
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        if (root == nullptr)
            return nullptr;

        if (root == p || root == q)
            return root;

        if (findNode(root->left, p) != nullptr
            && findNode(root->right, q) != nullptr)
            return root;
            
        if (findNode(root->right, p) != nullptr
            && findNode(root->left, q) != nullptr)
            return root;
        
        TreeNode* l = lowestCommonAncestor(root->left, p, q);
        if (l != nullptr)
            return l;
        TreeNode* r = lowestCommonAncestor(root->right, p, q);
        if (r != nullptr)
            return r;
        
        return nullptr;

    }
};
```



### (2)二叉搜索树的最近公共祖先

https://leetcode.cn/problems/lowest-common-ancestor-of-a-binary-search-tree/submissions/

```c++
class Solution {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        if (root == nullptr)
            return nullptr;

        if ((root->val > p->val && root->val < q->val)
            || (root->val > q->val && root->val < p->val))
            return root;
        
        if (root == p || root == q)
            return root;

        if (root->val > p->val)
            return lowestCommonAncestor(root->left, p, q);
        else // (root->val < p->val)
            return lowestCommonAncestor(root->right, p, q);
        
    }
};
```

### (2)删除二叉搜索树中的节点

https://leetcode.cn/problems/delete-node-in-a-bst/description/

```c++
class Solution {
public:
    TreeNode* findLeftNode(TreeNode* root){
        if (root == nullptr)
            return nullptr;
        
        while (root->left != nullptr)
            root = root->left;
        
        return root;
    }
    TreeNode* deleteNode(TreeNode* root, int key) {
        if (root == nullptr)
            return nullptr;
        
        if (key == root->val){
            if (root->left == nullptr && root->right == nullptr)
                return nullptr;
            if (root->right != nullptr){
                TreeNode* tmp = findLeftNode(root->right);
                root->val = tmp->val;
                root->right = deleteNode(root->right, tmp->val);
            }else if(root->left != nullptr)
                return root->left;
        }else if (key < root->val){
            root->left = deleteNode(root->left, key);
        }else{ // key > root->val
            root->right = deleteNode(root->right, key);
        }

        return root;

    }
};
```

```c++
        if (root == nullptr)
            return nullptr;
        
        root->left = trimBST(root->left, low, high);
        root->right = trimBST(root->right, low, high);

        if (root->val < low){
            return root->right;
        }else if (root->val > high){
            return root->left;
        }
        return root;
    }
```

```c++
class Solution {
public:
    TreeNode* func(vector<int>& nums, int left, int right){
        if (left > right)
            return nullptr;

        int midIdx = left + (right-left) / 2;
        TreeNode* root = new TreeNode(nums[midIdx], nullptr, nullptr);

        root->left = func(nums, left, midIdx-1);
        root->right = func(nums, midIdx+1, right);

        return root;
    }
    TreeNode* sortedArrayToBST(vector<int>& nums) {
        TreeNode* root = func(nums, 0, nums.size()-1);

        return root;
    }
};
```
