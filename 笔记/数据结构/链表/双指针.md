# 双指针

## 合并两个有序链表
```c++
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        ListNode* p1 = list1;
        ListNode* p2 = list2;

        ListNode* ans = new ListNode(-1, nullptr);
        ListNode* p = ans;
        while (p1 != nullptr || p2 != nullptr) {
            if (p1 != nullptr && (p2 == nullptr || p1->val < p2->val)) {
                // use list1's node
                p->next = p1;
                p = p->next;
                p1 = p1->next;
            } else if (p2 != nullptr && (p1 == nullptr || p1->val >= p2->val)) {
                p->next = p2;
                p = p->next;
                p2 = p2->next;
            }
        }
        return ans->next;
    }
};
```
