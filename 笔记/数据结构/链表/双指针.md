# 双指针

## 合并两个有序链表
```c++
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        ListNode* p1 = list1;
        ListNode* p2 = list2;

        ListNode* ans = new ListNode(-1, nullptr);
        ListNode* p = ans;
        while (p1 != nullptr || p2 != nullptr) {
            if (p1 != nullptr && (p2 == nullptr || p1->val < p2->val)) {
                // use list1's node
                p->next = p1;
                p = p->next;
                p1 = p1->next;
            } else if (p2 != nullptr && (p1 == nullptr || p1->val >= p2->val)) {
                p->next = p2;
                p = p->next;
                p2 = p2->next;
            }
        }
        return ans->next;
    }
};
```

## 分隔链表
```c++
// 两条链表，一条存储较大元素，一条存储较小元素
class Solution {
public:
    ListNode* partition(ListNode* head, int x) {
        if (head == nullptr)
            return head;
        
        ListNode* l1 = new ListNode(-1, nullptr);
        ListNode* l2 = new ListNode(-1, nullptr);

        ListNode* p1 = l1;
        ListNode* p2 = l2;
        ListNode* p = head;

        while (p != nullptr) {
            if (p->val < x) {
                p1->next = p;
                p1 = p1->next;
            } else if (p->val >= x) {
                p2->next = p;
                p2 = p2->next;
            }
            p = p->next;
        }

        p1->next = l2->next;
        p2->next = nullptr;
        
        return l1->next;
    }
};
```

## 合并k个升序链表
```c++
class Solution {
public:
    class MyCmp {
    public:
        bool operator() (ListNode* &lhs, ListNode* &rhs) {
            return lhs->val > rhs->val;
        }
    };
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        ListNode* dummy = new ListNode(-1, nullptr);
        ListNode* p = dummy;

        priority_queue<ListNode*, vector<ListNode*>, MyCmp> pq;

        for (ListNode* tmp : lists)
            if (tmp != nullptr)
                pq.push(tmp);

        while (! pq.empty()) {
            ListNode* minNode = pq.top();
            pq.pop();

            p->next = minNode;
            p = p->next;

            if (minNode->next != nullptr)
                pq.push(minNode->next);
        }

        return dummy->next;
    }
};
```
