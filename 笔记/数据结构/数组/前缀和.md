## 前缀和 + 哈希表
https://leetcode.cn/problems/subarray-sum-equals-k/submissions/533695952/?envType=study-plan-v2&envId=top-100-liked

```c++
class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        
        int ans = 0;
        vector<int> preSum(nums.size(), 0);

        // sum2cnt[i] = j, i = preSum, j = cnt
        unordered_map<int, int> sum2cnt;
        sum2cnt[0] = 1;

        for (int i = 0; i < nums.size(); i++) {
            if (i == 0) {
                preSum[0] = nums[0];
            } else {
                preSum[i] = preSum[i-1] + nums[i];
            }
            
            if (sum2cnt.count(preSum[i]-k) > 0) {
                ans += sum2cnt[preSum[i]-k];
            }

            if (sum2cnt.count(preSum[i]) == 0) {
                sum2cnt[preSum[i]] = 1;
            } else {
                sum2cnt[preSum[i]] += 1;
            }
        }
        
        return ans;
    }
};
```

```c#
public class Solution {
    public int SubarraySum(int[] nums, int k) {
        int preSum = 0;
        int ans = 0;
        Dictionary<int, int> sum2cnt = new Dictionary<int, int>();
        sum2cnt[0] = 1;
        for (int i = 0; i < nums.Length; i++) {
            preSum = preSum + nums[i];
            
            // preSum[i] - preSum[j] == k
            if (sum2cnt.ContainsKey(preSum-k)) {
                ans += sum2cnt[preSum-k];
            }

            if (sum2cnt.ContainsKey(preSum)) {
                sum2cnt[preSum] += 1;
            } else {
                sum2cnt[preSum] = 1;
            }
        }
        return ans;
    }
}
```
