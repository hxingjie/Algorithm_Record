正对角线交换：swap(matrix[i][j], matrix[j][i]);
逆对角线交换：swap(matrix[i][j], matrix[n-j-1][n-i-1]);
逆时针旋转：reverse -> 正对角线交换
顺时针旋转：正对角线交换 -> reverse

## 顺时针90度旋转数组
```c++
class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        int n = matrix.size();

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                swap(matrix[i][j], matrix[j][i]);
            }
        }

        for (int i = 0; i < n; i++) {
            int l = 0, r = n-1;
            while (l < r) {
                swap(matrix[i][l], matrix[i][r]);
                l += 1;
                r -= 1;
            }
        }
    }
};
```

## 螺旋遍历数组
```c#
public class Solution {
    public int[][] GenerateMatrix(int n)
    {
        int[][] ans = new int[n][];
        for (int i = 0; i < n; i++) ans[i] = new int[n];

        int rightBound = n - 1, lowerBound = n - 1, leftBound = 0, upperBound = 1;// 四个变量记录上下左右四个位置的边界值
        int row = 0, col = 0, val = 1;

        while (true)
        {
            for (; col <= rightBound; col++) ans[row][col] = val++;
            if (val > n * n) break;
            rightBound--;// 到达右边界，右边界收缩
            col--; row++;
                    
            for (; row <= lowerBound; row++) ans[row][col] = val++;
            if (val > n * n) break;
            lowerBound--;// 到达右边界，右边界收缩
            col--; row--;
                    
            for (; col >= leftBound; col--) ans[row][col] = val++;
            if (val > n * n) break;
            leftBound++;// 到达右边界，右边界收缩
            col++; row--;
                    
            for (; row >= upperBound; row--) ans[row][col] = val++;
            if (val > n * n) break;
            upperBound++;// 到达右边界，右边界收缩
            col++; row++;
        }

        return ans;
    }
}
```
