## 2.摆动序列

https://leetcode.cn/problems/wiggle-subsequence/

```c++
class Solution {
public:
    int wiggleMaxLength(vector<int>& nums) {
        int ans = 1;
        int last_diff = 0;
        for (int i = 1; i < nums.size(); i++){
            int diff = nums[i] - nums[i-1];
            
            if (diff == 0) continue;
            
            if (last_diff == 0 || last_diff * diff < 0){
                // last_diff == 0表示是第一个非平坡，不用和last_diff比较
                ans++;
                last_diff = diff;
            }

        }
        return ans;
    }
};
```



## 3.最大子数组和

https://leetcode.cn/problems/maximum-subarray/

```c++
class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        // dp[i]: 以nums[i]为结尾的连续子数组最大和
        // dp[i] = max(dp[i-1] + nums[i], nums[i])
        int ans = nums[0];
        vector<int> dp(nums.size(), 0);
        
        dp[0] = nums[0];
        for (int i = 1; i < nums.size(); i++){
            dp[i] = max(dp[i-1] + nums[i], nums[i]);
            ans = max(ans, dp[i]);
        }

        return ans;
    }
};
```



## 4.买卖股票的最佳时机II

https://leetcode.cn/problems/best-time-to-buy-and-sell-stock-ii/

```c++
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        // dp[i][j]: 第i天，j==1为持有，j==0为不持有
        // dp[i][0] = max(dp[i-1][0], dp[i-1][1]+prices[i])
        // dp[i][1] = max(dp[i-1][1], dp[i-1][0]-prices[i])

        int n = prices.size();
        vector<vector<int>> dp(n, vector<int>(2, 0));
        for (int i = 0; i < n; i++){
            if (i == 0){
                dp[0][0] = 0;
                dp[0][1] = -prices[0];
                continue;
            }
            dp[i][0] = max(dp[i-1][0], dp[i-1][1]+prices[i]);
            dp[i][1] = max(dp[i-1][1], dp[i-1][0]-prices[i]);
        }
        return dp[n-1][0];
    }
};
```



## 5.跳跃游戏

https://leetcode.cn/problems/jump-game/

```c++
class Solution {
public:
    bool canJump(vector<int>& nums) {
        int n = nums.size();
        
        int fastest_d = 0;
        for (int i = 0; i < n; i++){
            if (fastest_d < i) break;
            fastest_d = max(fastest_d, nums[i]+i);
        }
        return fastest_d >= n-1;
    }
};
```

```c++
class Solution {
public:
    int jump(vector<int>& nums) {
        int ans = 0;
        int n = nums.size();
        
        int fastest_d = 0;
        int end_idx = 0;

        for (int i = 0; i < n-1; i++){
            fastest_d = max(fastest_d, i + nums[i]);
            if (i == end_idx){
                ans++;// jump
                end_idx = fastest_d;
            }
        }
        return ans;
    }
```

## 7.K次取反后最大化的数组和

https://leetcode.cn/problems/maximize-sum-of-array-after-k-negations/

```c++
// 从小到大排序，先尽可能翻转负数，只剩下非负数时就重复翻转最小的非负数
// 优先翻转绝对值大的负数

class Solution {
public:
    int largestSumAfterKNegations(vector<int>& nums, int k) {
        sort(nums.begin(), nums.end());
        
        int cnt = 0;
        int minNum = INT_MAX;
        
        int i = 0;
        while (i < nums.size() && cnt < k){
            if (nums[i] < 0){
                nums[i] = -nums[i];
                cnt++;
            }
            minNum = min(minNum, nums[i]);
            i++;
        }
        
        int sum = 0;
        for (int j = 0; j < nums.size(); j++)
            sum += nums[j];
        
        // 翻转该数偶数次等于没有翻转，奇数次等于翻转一次
        if ((k-cnt) % 2 != 0)
            sum -= 2*minNum;// 3 - 6 == 3 + 6 - 2*6
        
        return sum;
    }
};
```



## 8.加油站

https://leetcode.cn/problems/gas-station/

```c++
// gas:    1 2 3 4 5
// cost:   3 4 5 1 2

// diff:  -2 -2 -2 3 3
// preSum: 0  2  4 6 3
// preSum[i]: 从i开始走到n-1所剩油量，即diff[i]+...+diff[n-1]
class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int ans = -1;
        int n = gas.size();

        vector<int> diff(n, 0);
        for (int i = 0; i < n; i++) diff[i] = gas[i] - cost[i];
            
        int maxVal = INT_MIN;
        vector<int> preSum(n, 0);
        for (int i = n-1; i >= 0; i--){
            if (i == n-1) preSum[i] = diff[i];
            else preSum[i] = preSum[i+1] + diff[i];

            if (preSum[i] > maxVal){
                maxVal = preSum[i];
                ans = i;
            } 
        }
        if (preSum[0] < 0) return -1;
            
        return ans;

    }
};
```



## 9.分发糖果

https://leetcode.cn/problems/candy/

```c++
class Solution {
public:
    // 1,2,87,87,87,2,1
    // 1 1  1  1  1 1 1

    // 1 0 2
    // 1 1 2
    int candy(vector<int>& ratings) {
        int n = ratings.size();

        vector<int> candies(n, 1);
        // 和左边的孩子比较, 左 -> 右,
        for (int i = 1; i <= n-1; i++){
            if (ratings[i] > ratings[i-1] 
                && candies[i] <= candies[i-1]){
                candies[i] = candies[i-1]+1;
            }
        }
        // 和右边的孩子比较, 左 <- 右
        for (int i = n-2; i >= 0; i--){
            if (ratings[i] > ratings[i+1]
                && candies[i] <= candies[i+1]){
                candies[i] = candies[i+1]+1;
            }
        }

        int sum = 0;
        for (int i = 0; i < candies.size(); i++)
            sum += candies[i];

        return sum;
    }
};
```



## 10.柠檬水找零

```c++
class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        vector<int> money = {0, 0, 0};// 5 10 20
        for (int i = 0; i < bills.size(); i++){
            int need = bills[i] - 5;
            if (need == 0){
                money[0]++;
            }else if (need == 5 && money[0] > 0){
                money[0]--;
                money[1]++;
            }else if (need == 15){
                if (money[0] >= 1 && money[1] >= 1){
                    // 贪心！优先使用15面额的零钱
                    money[0]--;
                    money[1]--;
                }else if (money[0] >= 3){
                    money[0] -= 3;
                }else{
                    return false;
                }
                money[2]++;
            }else{
                return false;
            }
        }

        return true;
    }
};
```



## 11.根据身高重建队列

https://leetcode.cn/problems/queue-reconstruction-by-height/

```c++
class Solution {
public:
    static bool cmp(const vector<int> &lhs, const vector<int> &rhs){
        if (lhs[0] > rhs[0]){
            return true;
        }else if (lhs[0] == rhs[0] && lhs[1] < rhs[1]){
            return true;
        }else{
            return false;
        }
    }
    vector<vector<int>> reconstructQueue(vector<vector<int>>& people) {
        sort(people.begin(), people.end(), cmp);

        list<vector<int>> l;
        for (vector<int> person : people){
            int k = person[1];
            list<vector<int>>::iterator it = l.begin();
            for (int i = 0; i < k; i++) it++;
            l.insert(it, person);
        }

        vector<vector<int>> q;
        list<vector<int>>::iterator it = l.begin();
        while (it != l.end()){
            q.push_back(*it);
            it++;
        }

        return q;
    }
};
```

### 12. 用最少数量的箭引爆气球

```c++
class Solution {
public:
    static bool cmp(const vector<int> &lhs, const vector<int> &rhs){
        if (lhs[1] < rhs[1] || (lhs[1] == rhs[1] && lhs[0] < rhs[0]))
            return true;
        else
            return false;
    }
    int findMinArrowShots(vector<vector<int>>& points) {
        sort(points.begin(), points.end(), cmp);
        
        int ans = 1;
        int curEnd = points[0][1];

        for (int i = 1; i < points.size(); i++){
            if (points[i][0] <= curEnd)
                continue;
            // points[i][0] > curEnd
            ans++;
            curEnd = points[i][1];
        }
        return ans;
    }
};
```



### 13.无重叠区间

https://leetcode.cn/problems/non-overlapping-intervals/

```c++
class Solution {
public:
    static bool cmp(const vector<int> &lhs, const vector<int> &rhs){
        if (lhs[1] < rhs[1] || (lhs[1] == rhs[1] && lhs[0] < rhs[0]))
            return true;
        else
            return false;
    }
    int eraseOverlapIntervals(vector<vector<int>>& intervals) {
        sort(intervals.begin(), intervals.end(), cmp);

        int ans = 0;
        int curEnd = intervals[0][1];

        for (int i = 1; i < intervals.size(); i++){
            if (intervals[i][0] < curEnd){// 重叠
                ans++;
                continue;
            }
            curEnd = intervals[i][1];// 不重叠，更新基准end
        }
        return ans;
    }
};
```



### 14.合并区间

https://leetcode.cn/problems/merge-intervals/submissions/497685994/

```c++
class Solution {
public:
    static bool cmp(const vector<int> &lhs, const vector<int> &rhs){
        if (lhs[0] < rhs[0] || (lhs[0] == rhs[0] && lhs[1] < rhs[1]))
            return true;
        else
            return false;
    }
    vector<vector<int>> merge(vector<vector<int>>& intervals) {
        vector<vector<int>> ans;

        sort(intervals.begin(), intervals.end(), cmp);

        int cur_beg = intervals[0][0];
        int cur_end = intervals[0][1];
        for (int i = 1; i < intervals.size(); i++){
            if (intervals[i][0] <= cur_end){
                cur_end = max(cur_end, intervals[i][1]);
            }else{
                vector<int> tmp = {cur_beg, cur_end};
                ans.push_back(tmp);
                cur_beg = intervals[i][0];
                cur_end = intervals[i][1];
            }
        }
        vector<int> tmp = {cur_beg, cur_end};
        ans.push_back(tmp);
        
        return ans;
    }
};
```



## 15.划分字母区间

https://leetcode.cn/problems/partition-labels/

```c++
class Solution {
public:
    vector<int> partitionLabels(string s) {
        // a b a b c b a c a  d  e  f  e  g  d  e
        // 8 4 8 4 7 4 8 7 8 14 15 11 15 13 14 15
        unordered_map<char, int> lastIdx;
        // 记录每个字符出现的最大下标
        for (int i = 0; i < s.size(); i++)
            lastIdx[s[i]] = i;
        
        vector<int> ans;
        int beg = 0;
        int maxIdx = -1;// 记录当前已遍历过的字符的最大下标
        for (int i = 0; i < s.size(); i++){
            int cur_lastIdx = lastIdx[s[i]];
            if (cur_lastIdx > maxIdx)
                maxIdx = lastIdx[s[i]];
            
            if (i == maxIdx){// [beg, i]即合法的最短字符串
                ans.push_back(maxIdx - beg + 1);
                beg = maxIdx + 1;
            }
        }

        return ans;
    }
};
```



## 16.单调递增的数字

https://leetcode.cn/problems/monotone-increasing-digits/

```c++
class Solution {
public:
    int monotoneIncreasingDigits(int n) {
        string num = to_string(n);

        int idx;// idx之后的数字(包括idx)都变为9
        for (int i = num.size()-1; i > 0; i--){
            if (num[i-1] > num[i]){// 小于右边的数
                num[i-1] -= 1;// 右边的数-1
                idx = i;// 该位之后的数包括该位变为9
            }
        }
        for (int i = idx; i < num.size(); i++)
            num[i] = '9';
        
        return stoi(num);
    }
};
```
